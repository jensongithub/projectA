select a.style_no,
       a.color_code,
       b.loc_code,
	   a.size_code,
	   sum(case when trans_type='A' or trans_type='I' then trans_qty else 0 end)- sum(case when trans_type='S' then trans_qty else 0 end)as 'qty',
	   sum(case when trans_type='O' then trans_qty else 0 end) as'qty1'
from tbl_pos_item_mstr  a join tbl_wh_stk_tx b
on  a.item_code=b.barcode_no
where 
isnull(b.rec_status,'A')='A' 
and b.comp_code='CAS'
and a.barcode_no = 'CAS044MANYCC0'
group by a.style_no,a.color_code,b.loc_code,a.size_code  
order by sum(case when trans_type='A' or trans_type='I' then trans_qty else 0 end)- sum(case when trans_type='S' then trans_qty else 0 end);
